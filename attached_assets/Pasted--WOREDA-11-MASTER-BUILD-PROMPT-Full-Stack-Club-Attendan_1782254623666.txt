# SECTION 3 — DATABASE SCHEMA

Run this SQL in Supabase SQL Editor:

```sql
-- ═══════════════════════════════════════════════════════════
-- WOREDA 11 — COMPLETE DATABASE SCHEMA
-- ═══════════════════════════════════════════════════════════

-- USERS TABLE
CREATE TABLE users (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  auth_id           UUID UNIQUE REFERENCES auth.users(id) ON DELETE CASCADE,
  username          TEXT UNIQUE NOT NULL,
  full_name         TEXT NOT NULL,
  email             TEXT UNIQUE NOT NULL,
  role              TEXT NOT NULL CHECK (role IN ('admin', 'manager')),
  club_id           UUID,
  profile_photo_url TEXT,
  is_active         BOOLEAN DEFAULT true,
  created_at        TIMESTAMPTZ DEFAULT now(),
  updated_at        TIMESTAMPTZ DEFAULT now()
);

-- CLUBS TABLE
CREATE TABLE clubs (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name        TEXT NOT NULL,
  icon_url    TEXT,
  manager_id  UUID REFERENCES users(id),
  is_active   BOOLEAN DEFAULT true,
  created_at  TIMESTAMPTZ DEFAULT now(),
  updated_at  TIMESTAMPTZ DEFAULT now()
);

-- Add club_id foreign key to users
ALTER TABLE users ADD CONSTRAINT users_club_id_fkey
  FOREIGN KEY (club_id) REFERENCES clubs(id) ON DELETE SET NULL;

-- MEMBERS TABLE
CREATE TABLE members (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  club_id      UUID NOT NULL REFERENCES clubs(id) ON DELETE CASCADE,
  full_name    TEXT NOT NULL,
  phone_number TEXT NOT NULL,
  age          INTEGER NOT NULL CHECK (age BETWEEN 5 AND 120),
  gender       TEXT NOT NULL CHECK (gender IN ('Male', 'Female', 'Other')),
  added_by     UUID REFERENCES users(id),
  created_at   TIMESTAMPTZ DEFAULT now(),
  updated_at   TIMESTAMPTZ DEFAULT now()
);

-- ATTENDANCE SESSIONS TABLE
CREATE TABLE attendance_sessions (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  club_id      UUID NOT NULL REFERENCES clubs(id) ON DELETE CASCADE,
  submitted_by UUID NOT NULL REFERENCES users(id),
  date         DATE NOT NULL,
  member_count INTEGER NOT NULL DEFAULT 0,
  submitted_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(club_id, date)
);

-- ATTENDANCE RECORDS TABLE
CREATE TABLE attendance_records (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id  UUID NOT NULL REFERENCES attendance_sessions(id) ON DELETE CASCADE,
  member_id   UUID NOT NULL REFERENCES members(id) ON DELETE CASCADE,
  attended_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(session_id, member_id)
);

-- NEWS TABLE
CREATE TABLE news (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title        TEXT NOT NULL,
  description  TEXT NOT NULL,
  image_url    TEXT,
  published_by UUID REFERENCES users(id),
  is_published BOOLEAN DEFAULT true,
  published_at TIMESTAMPTZ DEFAULT now(),
  updated_at   TIMESTAMPTZ DEFAULT now()
);

-- NEWS READS TABLE
CREATE TABLE news_reads (
  id      UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  news_id UUID NOT NULL REFERENCES news(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  read_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(news_id, user_id)
);

-- PUSH SUBSCRIPTIONS TABLE
CREATE TABLE push_subscriptions (
  id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id    UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  endpoint   TEXT NOT NULL,
  p256dh     TEXT NOT NULL,
  auth       TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(user_id, endpoint)
);

-- ═══════════════════════════════════════════════════════════
-- INDEXES
-- ═══════════════════════════════════════════════════════════
CREATE INDEX idx_members_club_id      ON members(club_id);
CREATE INDEX idx_sessions_club_id     ON attendance_sessions(club_id);
CREATE INDEX idx_sessions_date        ON attendance_sessions(date);
CREATE INDEX idx_records_session_id   ON attendance_records(session_id);
CREATE INDEX idx_records_member_id    ON attendance_records(member_id);
CREATE INDEX idx_news_reads_user      ON news_reads(user_id);
CREATE INDEX idx_push_subs_user       ON push_subscriptions(user_id);

-- ═══════════════════════════════════════════════════════════
-- ROW LEVEL SECURITY
-- ═══════════════════════════════════════════════════════════

ALTER TABLE users               ENABLE ROW LEVEL SECURITY;
ALTER TABLE clubs               ENABLE ROW LEVEL SECURITY;
ALTER TABLE members             ENABLE ROW LEVEL SECURITY;
ALTER TABLE attendance_sessions ENABLE ROW LEVEL SECURITY;
ALTER TABLE attendance_records  ENABLE ROW LEVEL SECURITY;
ALTER TABLE news                ENABLE ROW LEVEL SECURITY;
ALTER TABLE news_reads          ENABLE ROW LEVEL SECURITY;
ALTER TABLE push_subscriptions  ENABLE ROW LEVEL SECURITY;

-- Helper function: is current user admin?
CREATE OR REPLACE FUNCTION is_admin()
RETURNS BOOLEAN AS $$
  SELECT EXISTS (
    SELECT 1 FROM users
    WHERE auth_id = auth.uid() AND role = 'admin'
  );
$$ LANGUAGE sql SECURITY DEFINER;

-- Helper function: current user's club_id
CREATE OR REPLACE FUNCTION my_club_id()
RETURNS UUID AS $$
  SELECT club_id FROM users WHERE auth_id = auth.uid();
$$ LANGUAGE sql SECURITY DEFINER;

-- USERS policies
CREATE POLICY "users_read_own"   ON users FOR SELECT USING (auth_id = auth.uid());
CREATE POLICY "users_admin_all"  ON users FOR ALL    USING (is_admin());
CREATE POLICY "users_update_own" ON users FOR UPDATE USING (auth_id = auth.uid());

-- CLUBS policies
CREATE POLICY "clubs_admin_all"        ON clubs FOR ALL    USING (is_admin());
CREATE POLICY "clubs_manager_read_own" ON clubs FOR SELECT
  USING (manager_id IN (SELECT id FROM users WHERE auth_id = auth.uid()));

-- MEMBERS policies
CREATE POLICY "members_admin_all"    ON members FOR ALL USING (is_admin());
CREATE POLICY "members_manager_read" ON members FOR SELECT
  USING (club_id = my_club_id());
CREATE POLICY "members_manager_insert" ON members FOR INSERT
  WITH CHECK (club_id = my_club_id());
CREATE POLICY "members_manager_update" ON members FOR UPDATE
  USING (club_id = my_club_id());

-- ATTENDANCE SESSIONS policies
CREATE POLICY "sessions_admin_all"    ON attendance_sessions FOR ALL    USING (is_admin());
CREATE POLICY "sessions_manager_own"  ON attendance_sessions FOR SELECT
  USING (club_id = my_club_id());
CREATE POLICY "sessions_manager_insert" ON attendance_sessions FOR INSERT
  WITH CHECK (club_id = my_club_id());

-- ATTENDANCE RECORDS policies
CREATE POLICY "records_admin_all"   ON attendance_records FOR ALL USING (is_admin());
CREATE POLICY "records_manager_own" ON attendance_records FOR SELECT
  USING (session_id IN (
    SELECT id FROM attendance_sessions WHERE club_id = my_club_id()
  ));
CREATE POLICY "records_manager_insert" ON attendance_records FOR INSERT
  WITH CHECK (session_id IN (
    SELECT id FROM attendance_sessions WHERE club_id = my_club_id()
  ));

-- NEWS policies
CREATE POLICY "news_admin_all"      ON news FOR ALL    USING (is_admin());
CREATE POLICY "news_manager_read"   ON news FOR SELECT USING (is_published = true);

-- NEWS READS policies
CREATE POLICY "reads_own" ON news_reads FOR ALL USING (
  user_id IN (SELECT id FROM users WHERE auth_id = auth.uid())
);

-- PUSH SUBSCRIPTIONS policies
CREATE POLICY "push_own" ON push_subscriptions FOR ALL USING (
  user_id IN (SELECT id FROM users WHERE auth_id = auth.uid())
);

-- STORAGE BUCKETS

self.addEventListener('push', event => {
  const data = event.data?.json() ?? {}
  event.waitUntil(
    self.registration.showNotification(data.title || 'Woreda 11', {
      body:   data.body,
      icon:   data.icon || '/icon-192.png',
      badge:  '/badge-72.png',
      image:  data.image,
      data:   { url: data.url || '/' },
      tag:    data.tag || 'woreda11',
      actions: [
        { action: 'open',    title: 'Read Now' },
        { action: 'dismiss', title: 'Dismiss'  },
      ],
    })
  )
})

self.addEventListener('notificationclick', event => {
  event.notification.close()
  if (event.action === 'dismiss') return
  const url = event.notification.data?.url || '/'
  event.waitUntil(
    clients.matchAll({ type: 'window', includeUncontrolled: true })
      .then(list => {
        for (const client of list) {
          if ('focus' in client) return client.focus()
        }
        return clients.openWindow(url)
      })
  )
})

self.addEventListener('install',  e => e.waitUntil(self.skipWaiting()))
self.addEventListener('activate', e => e.waitUntil(self.clients.claim()))
```

# SECTION 24 — DATE/TIME RULES (STRICT)

All dates and times displayed in the app must follow these rules:

```ts
// src/lib/formatDate.ts

export function formatDate(iso: string): string {
  return new Intl.DateTimeFormat('en-GB', {
    day: '2-digit', month: 'long', year: 'numeric',
  }).format(new Date(iso))
  // Output: "14 January 2024"
}

export function formatShortDate(iso: string): string {
  return new Intl.DateTimeFormat('en-GB', {
    day: '2-digit', month: 'short', year: 'numeric',
  }).format(new Date(iso))
  // Output: "14 Jan 2024"
}

export function formatTime(iso: string): string {
  return new Intl.DateTimeFormat('en-GB', {
    hour: '2-digit', minute: '2-digit', hour12: true,
  }).format(new Date(iso))
  // Output: "02:35 PM"  — NO SECONDS EVER
}

export function formatDateTime(iso: string): string {
  return `${formatDate(iso)} at ${formatTime(iso)}`
  // Output: "14 January 2024 at 02:35 PM"
}

export function getTodayUTC(): string {
  return new Date().toISOString().slice(0, 10)
  // Output: "2024-01-14"  — always UTC DATE for DB
}

export function formatDayName(iso: string): string {
  return new Intl.DateTimeFormat('en-GB', { weekday: 'long' })
    .format(new Date(iso))
  // Output: "Tuesday"
}
```

Rules:
- **All dates: English Gregorian calendar**
- **All times: HH:MM AM/PM — never show seconds**
- **All text in English** — no other language
- **DB storage: always UTC**
- **Display: Intl.DateTimeFormat with en-GB locale**

# SECTION 27 — TOAST NOTIFICATION SYSTEM

```ts
// src/stores/notificationStore.ts
import { create } from 'zustand'

type Toast = {
  id: string
  type: 'success' | 'error' | 'info' | 'warning'
  title: string
  message?: string
  duration?: number
}

type BellNotif = {
  id: string
  type: 'success' | 'error' | 'info' | 'warning'
  title: string
  message: string
  url?: string
  read: boolean
  createdAt: string
}

export const useNotifStore = create((set, get) => ({
  toasts: [] as Toast[],
  bell:   [] as BellNotif[],

  addToast: (n: Omit<Toast, 'id'>) => {
    const id = crypto.randomUUID()
    set(s => ({ toasts: [...s.toasts, { ...n, id }].slice(-5) }))
    setTimeout(() => get().removeToast(id), n.duration ?? 5000)
  },

  removeToast: (id: string) =>
    set(s => ({ toasts: s.toasts.filter(t => t.id !== id) })),

  addBell: (n: Omit<BellNotif, 'id' | 'read' | 'createdAt'>) =>
    set(s => ({
      bell: [{ ...n, id: crypto.randomUUID(), read: false, createdAt: new Date().toISOString() },
             ...s.bell].slice(0, 50)
    })),

  markRead: (id: string) =>
    set(s => ({ bell: s.bell.map(b => b.id === id ? { ...b, read: true } : b) })),

  markAllRead: () => set(s => ({ bell: s.bell.map(b => ({ ...b, read: true })) })),
  unreadCount: () => (get() as any).bell.filter((b: BellNotif) => !b.read).length,
}))
```

## Toast Position
- Desktop: bottom-right
- Mobile: bottom-center (full width with margin)
- Max 5 stacked, FIFO
- Each: left-border accent color + title + message + X close + progress bar

export function exportMembers(members: any[], filename = 'members') {
  const rows = members.map((m, i) => ({
    '#':          i + 1,
    'Full Name':  m.full_name,
    'Age':        m.age,

    'Phone':      m.phone_number,
    'Club':       m.clubs?.name ?? '—',
    'Joined':     new Date(m.created_at).toLocaleDateString('en-GB'),
  }))

  const ws = XLSX.utils.json_to_sheet(rows)
  ws['!cols'] = [
    {wch:4},{wch:28},{wch:6},{wch:10},{wch:18},{wch:22},{wch:14}
  ]

  // Style header row
  const range = XLSX.utils.decode_range(ws['!ref'] ?? 'A1')
  for (let c = range.s.c; c <= range.e.c; c++) {
    const cell = XLSX.utils.encode_cell({ r: 0, c })
    if (!ws[cell]) continue
    ws[cell].s = {
      font: { bold: true, color: { rgb: 'FFFFFF' } },
      fill: { fgColor: { rgb: '255057' } },
    }
  }

  const wb = XLSX.utils.book_new()
  XLSX.utils.book_append_sheet(wb, ws, 'Members')
  XLSX.writeFile(wb, `woreda11_${filename}_${new Date().toISOString().slice(0,10)}.xlsx`)
}
